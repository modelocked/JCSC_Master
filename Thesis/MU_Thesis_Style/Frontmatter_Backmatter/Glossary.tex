\newglossaryentry{term:fpvd}
{
	name={First-Person View Drone (FPVD)},
	description={A category of UAS which is flown through goggles in the pilot's view. They have become synonymous with kamikaze and tactical OWA drones. However, FPVs can also function as CASDs --- with revolving drums utilised to produce `bombing runs'. For this reason, this thesis distinguishes between FPVDs broadly, FPVDs used as strike munitions (FPV-OWADs) and those providing close-air support (FPV-CASDs)}
}

\newglossaryentry{term:bai_def}
{
	name={Battlefield Air Interdiction (BAI)},
	description={\textcite[3-3--3-4]{usaf_1984_afm_1_1_basic_aerospace_doctrine}\footnote{See also \textcite{McCaffrey_2001}.} states that ``objectives are to delay, disrupt, divert, or destroy an enemy's military potential before it can be brought to bear effectively against friendly forces [;] Air interdiction attacks against targets which are in a position to have a near-term effect on friendly land forces are referred to as battlefield air interdiction''}
}

\newglossaryentry{term:isrd_def}
{
	name={Intelligence, Surveillance and Reconnaissance Drone (ISRD)},
	description={Fixed-wing or rotary-wing UAS which operate below the coordinating altitude. For the purposes of this thesis, rotary-wing UAS shall be inferred unless otherwise stated. While FPVs \textit{can} be employed, their continuous manual-piloting complicates their ISR applicability --- they cannot hover. Hence, assisted-flight UAS are the norm --- such as Parrot Anafi and DJI Mavic 3}
}

\newglossaryentry{term:rcp_def}
{
	name={Relative Combat Power (RCP)},
	description={It is a heuristic to assess military strength, analogous to the Lanchester Force Equations\index{Lanchester Force Equations}. It is assessed through a ``matrix providing detailed information on the relative combat power using factors across the physical, moral and conceptual frameworks''  \parencite[16]{L1} and \parencite[256]{LCH}, where common units and platforms are mathematically equated}
}


\newglossaryentry{term:ied_def}
{
	name={Improvised Explosive Device (IED)},
	description={``A device placed or fabricated in an improvised manner incorporating destructive, lethal, noxious, pyrotechnic or incendiary chemicals, designed to destroy, incapacitate, harass or distract'' \parencite[50]{DORD_2022}}
}

\newglossaryentry{term:voiedied_def}
{
	name={Victim-Operated IEDs (VOIEDs)},
	description={An IED initiated by the actions of the victim, typically through a pressure plate, tripwire or similar mechanism}
}

\newglossaryentry{dropper}
{
	name={Dropper Drone},
	description={Rotary-wing UAS (generally with a frame-size greater than $10\;''$) used to drop explosive payloads --- such as hand grenades, point-detonating ordnance and IEDs. Unlike FPV-OWADs, dropper drones are not consumable: they can be recovered, rearmed and re-tasked between sorties. Depending on control architecture, they may employ either manual first-person control (CAS-FPVD) or assisted-flight stabilisation, as with typical ISRDs. A dropper drone used for the close air support of frontline units is termed a Close Air Support Drone (CASD)}
}

	

\newglossaryentry{term:casd_def}
{
	name={Close Air Support Drones (CASD)},
	description={Dropper drones used for close air support of frontline units. Also referred to as a ``bomber drone''}
}

\newglossaryentry{term:owad_def}
{
	name={One Way Attack Drone (OWAD)},
	description={Long-range and slow-moving UAS intended to attack the enemy in depth and to supplement conventional cruise and ballistic missile fires \parencite{BRONK_2025}. \textcite{watling_2025_10} distinguishes them from FPVDs in that they are not actively piloted}
}

\newglossaryentry{doctrine}
{
	name={Doctrine},
	description={``Fundamental principles by which the military forces guide their actions in support of objectives. It is authoritative but requires judgement in application'' \parencite[K-3]{COPD}}
}

\newglossaryentry{payoff}
{
	name={High-Payoff Target},
	description={``A target of significance and value to an adversary, the destruction, damage or neutralisation of which may lead to a disproportionate advantage to friendly forces'' \parencite[K-4]{COPD}}
}

\newglossaryentry{transparent}
{
	name={Transparency (of the Battlefield)},
	description={``The ability to acquire and exploit geolocated, near-real-time awareness of a given operational environment thanks to a connectivity architecture linking networks of heterogeneous and redundant sensors, mass data processing systems, and effectors'' \parencite[3]{neron_bancel_2024_05_27}}
}

\newglossaryentry{positional}
{
	name={Positional Warfare},
	description={``Combined arms manoeuvre [...] whereby movement and fires undermine the enemy's ability to hold key terrain'' \parencite[1]{watling_2025_10}}
}

\newglossaryentry{attritional}
{
	name={Attritional Warfare},
	description={The [...] theory of attrition states that [belligerent] A seeks to change the relative balance of power in relation to [belligerent] B through the use of proximal and temporal firepower overmatch in consideration of both A's and B's battlefield disposition, for sustained periods of time, sufficient to gain its (A's) military objective \parencite[39]{FOX_2025}}
}

\newglossaryentry{killzone}
{
	name={Kill Zone},
	description={This is the section of the ``grey zone'' closest to the FLOT, where --- due to the porosity and dispersion of the defensive dispositions --- opposing positions are intermingled \parencite{grayzonekillzonet}. Its width varies --- reportedly being as wide as $10\;km$ in Ukraine. This will vary by conflict and is likely to be a function of the terrain}
}

\newglossaryentry{battledrill}
{
	name={Battle Drills},
	description={Pre-defined, rehearsed and well-understood actions or sequences executed in response to specific battlefield events. Their purpose is to reduce friction and enable rapid tactical action, minimising the need for further orders. They are triggered by the event itself rather than by a command}
}

\newglossaryentry{killchain}
{
	name={Kill-Chains},
	description={``Dynamic targeting is executed using the dynamic process of F2T2EA. [...] Targets of opportunity have been the traditional focus of dynamic targeting because decisions on whether and how to engage must be made quickly. [...] The steps of dynamic targeting may be accomplished iteratively and in parallel. The find, fix, track, and assess steps tend to be intelligence, surveillance, and reconnaissance intensive, while the target and engage steps are typically labor-, force-, and decision making-intensive.'' \parencite[II--23]{dod_2018_jp_3_60_joint_targeting}}
}

\newglossaryentry{greyzone}
{
	name={Grey Zone},
	description={Refers to the $20$--$30\;km$ area forward of the FLOT that has become increasingly dangerous to traverse due to persistent ISR and the prevalence of indirect and uncrewed fires \parencite{dan_participant_1,grayzonekillzonet}. This usage is distinct from the term as employed in international relations. The ``kill zone'' is a subset of the grey zone. It may also be described as the \textit{zone of contestation} or \textit{contested zone} \parencite[5--6]{watling_2025_10} \parencite[17:21--22:37]{WATLING_2023_book_launch}}
}

\newglossaryentry{term:ew_def}
{
	name={Electronic Warfare (EW)},
	description={Military action involving the use of EM energy and directed energy to control the EMS or to attack the enemy. EW consists of three divisions: electronic attack, electronic protection and electronic warfare support \parencite[viii]{joint_chiefs_2012_electronic_warfare}}
}

\newglossaryentry{elelcrecce}
{
	name={Electronic Reconnaissance (ER)},
	description={The detection, location, identification and evaluation of the adversary's EM radiations \parencite[I--9]{joint_chiefs_2012_electronic_warfare}}
}

\newglossaryentry{term:lrr_def}
{
	name={Long Range Reconnaissance (LRR)},
	description={``Extended-range reconnaissance, surveillance, target acquisition, and force protection over wider operational areas'' \parencite[28]{CALL_2025}}
}

\newglossaryentry{term:mrr_def}
{
	name={Medium Range Reconnaissance (MRR)},
	description={``Reconnaissance, surveillance, and target acquisition support for company- and battalion-level operations'' \parencite[28]{CALL_2025}}
}

\newglossaryentry{term:srr_def}
{
	name={Short Range Reconnaissance (SRR)},
	description={``Very short-range to short-range reconnaissance, direct situational awareness, and security for dismounted troops, squads, platoons, and companies'' \parencite{CALL_2025}}
}

\newglossaryentry{term:nai_def}
{
	name={Named Area of Interest (NAI)},
	description={A location identified during Intelligence Preparation of the Battlespace (IPB), derived from terrain analysis and assessment of the enemy's doctrine and likely courses of action. NAIs are areas assessed as potentially significant for observation in order to confirm or deny enemy activity. The number of NAIs will typically exceed available collection assets; consequently, not all NAIs can be observed at any given time}
}

\newglossaryentry{term:tai_def}
{
	name={Target Area of Interest (TAI)},
	description={A Named Area of Interest (NAI) where fires must be delivered in accordance with the targeting plan. TAIs are linked to decision points and enable the rapid engagement of identified enemy forces}
}

\newglossaryentry{term:suas_def}
{
	name={Small Uncrewed Aerial System (sUAS)},
	description={A UAS with a maximum take-off weight below a defined threshold --- typically $25\;kg$ --- employed for tactical reconnaissance, surveillance and target acquisition at sub-company level}
}

\newglossaryentry{class}
{
	name={UAS Classes},
	description={There is no agreed-upon standard for classifying UAS. Characteristics such as mass, range and ceiling are often used. This thesis concurs with \textcite[5]{BRONK_2024} and considers five classes: ``Situational awareness UAVs optimised for tactical reconnaissance; tactical strike UAVs; ISR UAVs able to penetrate into operational depth; operational strike UAVs; and platform-launched effects designed specifically to synchronise with and enable other weapons systems''}
}

\newglossaryentry{waiters}
{
	name={Waiter Drones (waiters)},
	description={The pre-deployment of a monitored FPVD to a location in anticipation of enemy activity. Modified with skids so that it can land and take off, allowing it to lie in wait, the drone is activated and attacks when a target presents; when none presents, it rests passively on the ground awaiting RC instructions. Commonly referred to as a ``waiter'', ``sleeper'' or ``ambush'' drone}
}

\newglossaryentry{hunter-killer}
{
	name={Hunter Killer Drone},
	description={An FPVD flown into the battlespace to search actively for enemy activity and engage suitable targets of opportunity --- as distinct from the pre-positioned waiter. Such drones typically use an improvised point-detonating fuze and/or RC initiation; a typical point-detonating fuze comprises an open switch of two bare wire loops on the drone's nose which close on impact}
}
